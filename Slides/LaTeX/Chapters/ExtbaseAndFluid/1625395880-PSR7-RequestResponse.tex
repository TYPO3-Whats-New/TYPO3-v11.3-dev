% Feature | 94428 | Extbase Request implements ServerRequestInterface

\begin{frame}[fragile]
	\frametitle{Extbase und Fluid}
	\framesubtitle{PSR-7 Anfrage/Antwort (1)}

	\begin{itemize}
		\item Um einheitliche Standards und modernste Technologien für
			Extension-Entwickler zu gewährleisten, sollte Extbase ein sauberes
			Request/Response-Handling nach dem
			\href{https://www.php-fig.org/psr/psr-7/}{PSR-7 Standard} bieten
		\item Daher implementiert die Klasse
			\small\texttt{TYPO3\textbackslash
				CMS\textbackslash
				Extbase\textbackslash
				Mvc\textbackslash
				Request}\normalsize\newline
			nun das PSR-7
			\texttt{ServerRequestInterface}
		\item Damit können Entwickler auf Request-bezogene Informationen aus dem Core zugreifen
		\item Die Schnittstelle ist zum Beispiel innerhalb von Extbase-Controllern verfügbar:\newline
			\texttt{\$this->request}
		\item Siehe das
			\href{https://docs.typo3.org/c/typo3/cms-core/master/en-us/Changelog/11.3/Feature-94428-ExtbaseRequestImplementsServerRequestInterface.html}{changelog}
			für weitere Details
	\end{itemize}

\end{frame}
