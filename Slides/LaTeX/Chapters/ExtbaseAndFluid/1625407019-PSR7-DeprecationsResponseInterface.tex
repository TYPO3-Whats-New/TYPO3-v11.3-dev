% Deprecation | 94351 | ext:extbase StopActionException

\begin{frame}[fragile]
	\frametitle{Extbase und Fluid}
	\framesubtitle{PSR-7 Anfrage/Antwort (2)}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Extbase-Aktionen \textbf{müssen} in Zukunft ein \texttt{ResponseInterface}
			zurückgeben
		\item Erweiterungen sind betroffen, wenn:
			\begin{itemize}
				\item sie eine \texttt{StopActionException} auslösen
				\item sie die Methode \texttt{redirect()} verwenden
				\item sie die Methode \texttt{redirectToUri()} verwenden
			\end{itemize}
		\item Beispiel für die Migration:
\begin{lstlisting}
// Code before:
public function fooAction()
{
   $this->redirect('otherAction');
}

// Code after:
public function fooAction(): ResponseInterface
{
   return $this->redirect('otherAction');
}
\end{lstlisting}

	\end{itemize}

\end{frame}
