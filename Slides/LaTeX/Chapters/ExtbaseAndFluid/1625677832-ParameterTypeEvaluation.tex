% Deprecation | 94115 | Parameter Type Evaluation via DocBlock comments

\begin{frame}[fragile]
	\frametitle{Extbase und Fluid}
	\framesubtitle{Parameter Typ Auswertung}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{7}{9}\ttfamily}

	\begin{itemize}
		\item Extbase's \textit{Legacy}-Typ-Erkennung von Argumenten
			wurde als \textbf{veraltet} markiert
		\item Entwickler sollten PHP's Typ-Hinweise in Argumentlisten verwenden
		\item Beispiel (\textbf{alter} Code):
\begin{lstlisting}
/**
 * @param \MyVendor\MyExtension\MyModel $item
 */
public function myAction($item);
\end{lstlisting}
			\textbf{neuer} Code:
\begin{lstlisting}
public function myAction(\MyVendor\MyExtension\MyModel $item);
\end{lstlisting}

	\end{itemize}

\end{frame}
