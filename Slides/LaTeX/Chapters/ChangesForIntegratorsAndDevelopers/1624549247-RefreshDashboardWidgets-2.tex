% Feature | 93210 | Possibility to refresh dashboard widgets

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{Dashboard-Widgets aktualisieren (2)}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{8}{10}\ttfamily}

	\begin{itemize}
		\item Setzen Sie die Konfigurationsoption \texttt{refreshAvailable} auf \texttt{true}
		\item Beispiel YAML-Konfiguration:
\begin{lstlisting}
dashboard.widget.foobarWidget:
  class: 'TYPO3\CMS\Dashboard\Widgets\RssWidget'
  arguments:
    $view: '@dashboard.views.widget'
    $cache: '@cache.dashboard.rss'
    $options:
      rssFile: 'https://example.com/rss'
      lifeTime: 3600
      refreshAvailable: true
  tags:
    ...
\end{lstlisting}

	\end{itemize}

\end{frame}
