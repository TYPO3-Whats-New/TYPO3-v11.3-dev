% Important | 91496 | Changes to Password Reset functionality

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{Zurücksetzen des Passworts}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{8}{10}\ttfamily}

	\begin{itemize}
		\item Seit TYPO3 v10 LTS deaktiviert die folgende globale Konfiguration die
			Passwort-Reset-Funktion für Backend-Benutzer:
\begin{lstlisting}
$GLOBALS['TYPO3_CONF_VARS']['BE']['passwordReset'] = false
\end{lstlisting}

		\item Änderungen in TYPO3 v11.3:

		 	\begin{itemize}
				\item Die Konfiguration beeinflußt nur den Anmeldebildschirm
				\item Administratoren können nach wie vor einen Passwort-Reset für Benutzer über das Backend durchführen
				\item Administratoren können weiterhin den entsprechenden CLI-Befehl ausführen
			\end{itemize}

		\item Eine neue User TSconfig Option deaktiviert die Funktion im Backend:
\begin{lstlisting}
options.passwordReset = 0
\end{lstlisting}

	\end{itemize}

\end{frame}
