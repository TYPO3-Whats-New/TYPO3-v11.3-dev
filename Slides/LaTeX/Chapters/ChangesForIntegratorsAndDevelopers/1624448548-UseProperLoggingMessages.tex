% Important | 94315 | Use proper PSR-3 logging messages and context

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{PSR-3 Logging-Meldungen und Kontext}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{8}{10}\ttfamily}

	\begin{itemize}
		\item Um die \href{https://www.php-fig.org/psr/psr-3/}{PSR-3 Spezifikation} vollständig zu erfüllen,
			verwendet der TYPO3 Core nun Platzhalter bei der Ausführung von Logging-Meldungen:
\begin{lstlisting}
$this->logger->alert(
  'Text text text {placeholder} text text.',
  ['placeholder' => $variable]
);
\end{lstlisting}

		\item Wir empfehlen den Entwicklern von Erweiterungen, ihren Code zu
			überprüfen und die Spezifikation zu befolgen
		\item Implementieren Sie die Protokollierung \textbf{nicht} auf diese Weise:
\begin{lstlisting}
$this->logger->alert('Text text text ' . $variable . ' text text.');
\end{lstlisting}

	\end{itemize}

\end{frame}
