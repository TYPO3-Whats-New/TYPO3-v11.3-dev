% Feature | 92358 | Add getModuleTemplate() to PageLayoutController

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{PageLayoutController (1)}

	\begin{itemize}
		\item Der \texttt{PageLayoutController} enthält nun die neue Methode\newline
			\texttt{getModuleTemplate()}
		\item Dies gibt Entwicklern mehr Möglichkeiten, das Seitenmodul zu manipulieren,
			wenn sie den \texttt{drawHeaderHook} und \texttt{drawFooterHook} verwenden
		\item Die neue Methode kann verwendet werden, um die entsprechende ModuleTemplate
			Instanz abzurufen, wie der Beispielcode auf der nächsten Seite zeigt
	\end{itemize}

\end{frame}
