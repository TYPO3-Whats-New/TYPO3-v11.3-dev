% Feature | 93668 | Possibility to configure Symfony mailer

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{Symfony Mailer}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{8}{10}\ttfamily}

	\begin{itemize}
		\item Integratoren können jetzt den Symfony Mailer mit DSN konfigurieren:
\begin{lstlisting}
[MAIL][transport] = 'dsn'
\end{lstlisting}
		\item Dies macht es möglich, jeden 3rd-Party Mail-Transport-Anbieter zu verwenden,
			wie Amazon SES, Gmail, MailChimp und viele mehr
		\item Konfigurationsbeispiel (Amazon SES):
\begin{lstlisting}
[MAIL][dsn] = 'ses+api://ACCESS_KEY:SECRET_KEY@default'
\end{lstlisting}

		\item Die Abhängigkeiten müssen mit Composer installiert werden
		\item Siehe \href{https://symfony.com/doc/current/mailer.html}{Symfony Dokumentation}
			für weitere Details über verfügbare Anbieter
	\end{itemize}

\end{frame}
