% Feature | 94206 | Add excludePagesRecursive option to XML sitemap generation

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{XML-Sitemap}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{6}{8}\ttfamily}

	\begin{itemize}
		\item Eine neue TypoScript-Konfiguration \texttt{excludePagesRecursive}
			wurde hinzugefügt, um Teile des Seitenbaums von der Sitemap auszuschließen
		\item Die Option akzeptiert eine kommagetrennte Liste von Seiten-IDs
		\item Zum Beispiel:
\begin{lstlisting}
plugin.tx_seo {
  config {
    xmlSitemap {
      sitemaps {
        pages {
          config {
            # exclude pages recursively
            excludePagesRecursive = 2,3
          }
        }
      }
    }
  }
}
\end{lstlisting}

	\end{itemize}

\end{frame}
