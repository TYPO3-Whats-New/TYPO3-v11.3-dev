% Feature | 92518 | Download and filename options added to FileDumpController

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{FileDumpController}

	\begin{itemize}
		\item Der \texttt{FileDumpController} akzeptiert zwei neue Parameter:
			\begin{itemize}
				\item \texttt{dl}: erzwingt das Herunterladen einer Datei
				\item \texttt{fn}: Herunterladen der Datei mit einem alternativen Dateinamen
			\end{itemize}
		\item Siehe das
			\href{https://docs.typo3.org/c/typo3/cms-core/master/en-us/Changelog/11.3/Feature-92518-DownloadAndFilenameOptionsAddedToFileDumpController.html}{changelog}
			für weitere Details und ein Code-Beispiel
		\item Ein neuer \textit{File}-ViewHelper kann verwendet werden, um die Funktion zu nutzen\newline
			\small(siehe Kapitel "Extbase und Fluid" für weitere Details)\normalsize

	\end{itemize}

\end{frame}
