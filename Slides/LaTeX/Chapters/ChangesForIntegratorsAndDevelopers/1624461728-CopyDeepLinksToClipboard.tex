% Feature | 93921 | Sharing backend links

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{Teilen von Backend-Links}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{8}{10}\ttfamily}

	\begin{itemize}
		\item Eine neue Konstante \texttt{SHAREABLE\_URL} ist im UriBuilder verfügbar
		\item Entwickler können mit dieser Konstante Deep Links (URLs) von Seiten im
			Backend erzeugen
		\item Zum Beispiel:
\begin{lstlisting}
$uri = $uriBuilder->buildUriFromRoute(
  $routeName, $arguments, UriBuilder::SHAREABLE_URL
);
\end{lstlisting}

	\end{itemize}

\end{frame}
