% Feature | 94081 | TCA readOnly for t3editor

\begin{frame}[fragile]
	\frametitle{Änderungen für Integratoren und Entwickler}
	\framesubtitle{TCA: \texttt{t3editor}}

	% decrease font size for code listing
	\lstset{basicstyle=\fontsize{8}{10}\ttfamily}

	\begin{itemize}
		\item Eine neue TCA-Option \texttt{readOnly} wurde dem Rendertyp "t3editor"
			hinzugefügt, um zu verhindern, dass Benutzer den Inhalt bearbeiten
		\item TCA-Code-Beispiel:
\begin{lstlisting}
'config' => [
  'type' => 'text',
  'renderType' => 't3editor',
  'format' => 'html',
  'readOnly' => true
]
\end{lstlisting}

	\end{itemize}

\end{frame}
