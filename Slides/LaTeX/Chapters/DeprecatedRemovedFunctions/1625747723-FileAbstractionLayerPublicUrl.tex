% Deprecation | 94193 | Public url with relative paths in FAL API

\begin{frame}[fragile]
	\frametitle{Veraltete/entfernte Funktionen}
	\framesubtitle{\texttt{getPublicUrl()} auf FAL-Objekten}

	\begin{itemize}
		\item Beim Zugriff auf \texttt{getPublicUrl()} auf ein FAL-Objekt konnte der
			relative Pfad durch Setzen der folgenden Variablen abgerufen werden:\newline
			\small\texttt{\$relativeToCurrentScript = true}\normalsize
		\item Da dies nicht mehr erforderlich ist (siehe weitere Details im
			\href{https://docs.typo3.org/c/typo3/cms-core/master/en-us/Changelog/11.3/Deprecation-94193-PublicUrlWithRelativePathsInFALAPI.html}{changelog}),
			wurde dies als \textbf{veraltet} markiert
		\item Der Zugriff auf \texttt{isRelativeToCurrentScript()} im
			\texttt{GeneratePublicUrlForResourceEvent} ist ebenfalls als
			\textbf{veraltet} markiert
		\item Der
			\href{https://docs.typo3.org/m/typo3/reference-coreapi/master/en-us/ApiOverview/ExtensionScanner/Index.html}{Extension Scanner}
			von TYPO3 erkennt die Verwendung dieser FÃ¤lle

	\end{itemize}
