% Deprecation | 94137 | Switch behavior of ArrayUtility::arrayDiffAssocRecursive()

\begin{frame}[fragile]
	\frametitle{Veraltete/entfernte Funktionen}
	\framesubtitle{Methode \texttt{arrayDiffAssocRecursive()}}

	\begin{itemize}
		\item Das Verhalten der folgenden Methode hat sich geändert:\newline
			\smaller\texttt{TYPO3\textbackslash
				CMS\textbackslash
				Core\textbackslash
				Utility\textbackslash
				ArrayUtility::arrayDiffAssocRecursive()}\normalsize
		\item Die Funktion imitiert nun das Verhalten der PHP-Funktion
			\texttt{array\_diff\_assoc()}, wie der Name schon sagt
		\item Dies muss durch Setzen des dritten Parameters auf
			\texttt{true} für die Abwärtskompatibilität
			(\texttt{\$useArrayDiffAssocBehavior}) aktiviert werden
		\item Entwickler, die das Verhalten der PHP-Funktion
			\texttt{array\_diff\_key()} benötigen, sollten die neue Methode
			\texttt{arrayDiffKeyRecursive()} der gleichen Klasse verwenden
	\end{itemize}

\end{frame}
