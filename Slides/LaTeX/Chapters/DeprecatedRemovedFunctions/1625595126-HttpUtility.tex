% Deprecation | 94316 | Deprecated HTTP header manipulating methods from HttpUtility

\begin{frame}[fragile]
	\frametitle{Deprecated/Removed Functions}
	\framesubtitle{HTTP Utility}

	\begin{itemize}
		\item To properly follow the \href{https://www.php-fig.org/psr/psr-7/}{PSR-7 standard},
			the following methods from the \textt{HttpUtility} have been marked
			as \textbf{deprecated}:

			\begin{itemize}
				\item \texttt{redirect()}
				\item \texttt{setResponseCode()}
				\item \texttt{setResponseCodeAndExit()}
			\end{itemize}

	\end{itemize}

\end{frame}
