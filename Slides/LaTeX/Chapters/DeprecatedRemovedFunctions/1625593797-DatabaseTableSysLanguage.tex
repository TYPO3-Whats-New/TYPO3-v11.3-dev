% Deprecation | 94165 | sys_language table

\begin{frame}[fragile]
	\frametitle{Veraltete/Entfernte Funktionen}
	\framesubtitle{Datenbanktabelle \texttt{sys\_language}}

	\begin{itemize}
		\item Die Verwendung der Datenbanktabelle \texttt{sys\_language} wurde
			als \textbf{veraltet} eingestuft
		\item Site-Sprachen, die in Site-Konfigurationen verwendet werden,
			sind jetzt unabhängig von jedem \texttt{sys\_language}-Datensatz
		\item Entwicklern wird empfohlen, die verfügbaren Sprachen und die damit
			verbundenen Informationen aus der Site-Konfiguration zu lesen
		\item Verwenden Sie zum Beispiel den neuen TCA-Typ \texttt{language} anstelle von
			\texttt{foreign\_table => sys\_language}
			(siehe \href{https://forge.typo3.org/issues/57082}{\#57082} für Details)
	\end{itemize}

\end{frame}
