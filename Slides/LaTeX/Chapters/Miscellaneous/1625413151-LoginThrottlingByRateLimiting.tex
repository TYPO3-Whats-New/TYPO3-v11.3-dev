\begin{frame}[fragile]
	\frametitle{Sonstiges}
	\framesubtitle{Login-Drosselung}

	\begin{itemize}
		\item TYPO3 begrenzt jetzt Backend- und Frontend-Logins mit der Symfony-Bibliothek
			"\href{https://symfony.com/doc/current/rate_limiter.html}{Rate Limiter}"
		\item Eine PSR-15-Middleware verwendet nun das Ratenlimit, um weitere Authentifizierungsversuche
			zu blockieren, wenn zu viele fehlgeschlagene Anmeldungen von derselben IP-Adresse festgestellt wurden
		\item Dieser Ansatz ersetzt die alte Lösung, die unter Sicherheitsaspekten
			nicht dem neuesten Stand der Technik entsprach.
		\item Siehe das
			\href{https://docs.typo3.org/c/typo3/cms-core/master/en-us/Changelog/11.3/Feature-93825-IntroduceRateLimiterComponent.html}{changelog}
			 für Konfigurationsoptionen, Standardwerte und weitere Details
	\end{itemize}
\end{frame}
