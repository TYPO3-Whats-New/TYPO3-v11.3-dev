% Feature | 93114 | Native Support for Language Shona (Bantu) added
% Feature | 94447 | Native Support for Language Welsh added

\begin{frame}[fragile]
	\frametitle{Sonstiges}
	\framesubtitle{Sprachunterstützung}

	\begin{itemize}
		\item Das TYPO3-Backend unterstützt jetzt zwei weitere Sprachen:
			\begin{itemize}
				\item \href{https://en.wikipedia.org/wiki/Shona_language}{Shona (Bantu)}
					(ISO 639-1 code: \texttt{sn})
				\item \href{https://en.wikipedia.org/wiki/Welsh_language}{Walisisch}
					(ISO 639-1 code: \texttt{cy})
			\end{itemize}
			\vspace{0.2cm}
		\item Diese Änderung ermöglicht es uns:
			\begin{itemize}
				\item übersetzte Labels für diese Sprachen von\newline
					\href{https://translation.typo3.org}{translation.typo3.org}
					und
					\href{https://crowdin.com}{CrowdIn} abzurufen
				\item die Backend-Schnittstelle auf diese Sprachen umzustellen
				\item diese Sprachen in einer Website-Konfiguration zu verwenden
				\item Übersetzungsdateien zu erstellen (z.B. \texttt{<code>.locallang.xlf})
			\end{itemize}

	\end{itemize}

\end{frame}
