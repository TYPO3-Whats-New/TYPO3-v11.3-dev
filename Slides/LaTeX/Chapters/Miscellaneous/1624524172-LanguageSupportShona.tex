% Feature | 93114 | Native Support for Language Shona (Bantu) added

\begin{frame}[fragile]
	\frametitle{Miscellaneous}
	\framesubtitle{Language Support}

	\begin{itemize}
		\item The TYPO3 backend now supports the
			\href{https://en.wikipedia.org/wiki/Shona_language}{Shona (Bantu)} language
		\item Shona is the language of the Shona people of Zimbabwe
		\item ISO 639-1 code: \texttt{sn}
		\item This change enables us to:
			\begin{itemize}
				\item Fetch translated labels from
					\href{https://translation.typo3.org}{translation.typo3.org}
					and
					\href{https://crowdin.com}{CrowdIn}
				\item Switch the Backend interface to the Shona language
				\item Use this language in a site configuration
				\item Create translation files, for example: \texttt{sn.locallang.xlf}
			\end{itemize}

	\end{itemize}

\end{frame}
